document.addEventListener("DOMContentLoaded", function () {
  const video = document.getElementById("heroVideo");
  const box = document.getElementById("videoBox");
  const icon = document.getElementById("videoIcon");
  const playIcon = document.getElementById("playIcon");
  const pauseIcon = document.getElementById("pauseIcon");

  const volumeBtn = document.getElementById("volumeBtn");
  const volumeIcon = document.getElementById("volumeIcon");
  const progressBar = document.getElementById("progressBar");
  const timeLabel = document.getElementById("timeLabel");
  const fullscreenBtn = document.getElementById("fullscreenBtn");

  // 1) click anywhere to play/pause
  box.addEventListener("click", function (e) {
    // controls e click korle toggle hobe na
    if (e.target.closest(".video-controls")) return;

    togglePlay();
  });

  function togglePlay() {
    if (video.paused) {
      video.play();
      showIcon("play");
    } else {
      video.pause();
      showIcon("pause");
    }
  }

  function showIcon(type) {
    playIcon.style.display = "none";
    pauseIcon.style.display = "none";

    if (type === "play") {
      playIcon.style.display = "block";
    } else {
      pauseIcon.style.display = "block";
    }

    icon.classList.add("show");
    setTimeout(() => {
      icon.classList.remove("show");
    }, 500);
  }

  // 2) volume toggle (svg)
  volumeBtn.addEventListener("click", function (e) {
    e.stopPropagation();
    video.muted = !video.muted;

    if (video.muted) {
      volumeIcon.src = "Images/Icons/Mute.svg";
      volumeIcon.alt = "Muted";
    } else {
      volumeIcon.src = "Images/Icons/Volume.svg";
      volumeIcon.alt = "Volume On";
    }
  });

  // 3) time/progress update
  video.addEventListener("timeupdate", function () {
    if (!video.duration) return;
    const percent = (video.currentTime / video.duration) * 100;
    progressBar.value = percent;
    updateTimeLabel();
  });

  // 4) seek
  progressBar.addEventListener("input", function (e) {
    if (!video.duration) return;
    const seekTime = (progressBar.value / 100) * video.duration;
    video.currentTime = seekTime;
  });

  // 5) fullscreen
  fullscreenBtn.addEventListener("click", function (e) {
    e.stopPropagation();
    if (!document.fullscreenElement) {
      box.requestFullscreen && box.requestFullscreen();
    } else {
      document.exitFullscreen && document.exitFullscreen();
    }
  });

  // 6) keyboard shortcuts
  document.addEventListener("keydown", function (e) {
    // input, textarea, contenteditable hole ignore
    const tag = e.target.tagName.toLowerCase();
    if (tag === "input" || tag === "textarea") return;

    switch (e.key.toLowerCase()) {
      case " ": // space
      case "k":
        e.preventDefault();
        togglePlay();
        break;
      case "m":
        video.muted = !video.muted;
        if (video.muted) {
          volumeIcon.src = "Images/Icons/Mute.svg";
        } else {
          volumeIcon.src = "Images/Icons/Volume.svg";
        }
        break;
      case "f":
        if (!document.fullscreenElement) {
          box.requestFullscreen && box.requestFullscreen();
        } else {
          document.exitFullscreen && document.exitFullscreen();
        }
        break;
      case "arrowright":
        video.currentTime = Math.min(video.currentTime + 5, video.duration || video.currentTime);
        break;
      case "arrowleft":
        video.currentTime = Math.max(video.currentTime - 5, 0);
        break;
    }
  });

  // helper: time format
  function updateTimeLabel() {
    const cur = formatTime(video.currentTime);
    const dur = formatTime(video.duration);
    timeLabel.textContent = `${cur} / ${dur}`;
  }

  function formatTime(sec) {
    if (isNaN(sec)) return "0:00";
    const m = Math.floor(sec / 60);
    const s = Math.floor(sec % 60);
    return m + ":" + (s < 10 ? "0" + s : s);
  }

  // video load hole first time update
  video.addEventListener("loadedmetadata", updateTimeLabel);
});
